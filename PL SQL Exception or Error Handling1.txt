	Continue to our session with same file name now we are going to find out the errors during query execution

	Consider Followin example we have 	

	declare
	name emp.emp_name%type;
	st varchar2(1000);
	begin
	-- Correct column name
	st:='select ' || ' emp_name ' || 'from emp where rownum = 1';
	execute immediate st into name;
	dbms_output.put_line(name);
	exception
	when others then
	dbms_output.put_line('SQLCODE: '|| SQLCODE);
	dbms_output.put_line('SQLERRM: '|| SQLERRM);
	end;


	Output will be the result fetch by block
